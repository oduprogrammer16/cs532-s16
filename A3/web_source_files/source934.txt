function getSiteWidth( w ) {	
	var width = w >= 1280 ? 1280 :
	                w >= 980 ? 980 : 
	                w >= 768 ? 768 : 
	                w >= 480 ? 480 : 320;
	return width;
}

function adjustLayout(force) {	
	var width = $(window).width();
	var win_h = $(window).height();
	var hero_width = 1280;
	var hero_height = 494;
	
	width = getSiteWidth( width );
	if ( force || (width != CWTV.Site.width)) {
		var bannerResize = (CWTV.Site.width < 980 && width >= 980)
					|| (CWTV.Site.width >= 980 && width < 980);
		if ( bannerResize ) {
			reloadBannerSlots();
		}
		CWTV.Site.width = width;
						
		var onMobileScreen = (width < 480) || CWTV.Site.ua.is_ios;
		hero_width = CWTV.Site.width;

		var new_class = (CWTV.Site.width >= 1280) ? 'wLarge' :
		                (CWTV.Site.width <= 320) ? 'wSmall' : 'w'+CWTV.Site.width;
	    $('body').removeClass('wLarge w980 w768 w480 wSmall').addClass(new_class);
		
		$("#full-wrapper").trigger("windowresize");
		
		if ( !CWTV.Site.overlayIsOpen ) {

			$("#herolist .heroitem, #herolist .heroimg").css('width', hero_width+'px').css('height', hero_height+'px');
			if (!onMobileScreen) {
				$("#herocontainer").css('width', hero_width+'px');
				$("#herocontainer .touchcarousel-wrapper").css('height', hero_height+'px');
			}
			else
				$("#herocontainer").css('height', 'auto');

			if ( CWTV.Site.heroSliderInstance ) {
				//$("#herocontainer").trigger(CWTV.Site.heroSliderInstance._resizeEvent);
			}
			else {
				initHero( hero_width );
			}

			var thumb_width = 320;
			var itemsVisible = Math.floor(CWTV.Site.width / thumb_width);

			if ( CWTV.Site.newVideosSliderInstance ) {
				var latestItemsVisible = itemsVisible - 1;
				var latest_videos_width = latestItemsVisible * thumb_width;
				$("#homevideolatest-wrap").css('width', latest_videos_width+'px');
			}

			var videos_thumbs_width = itemsVisible * thumb_width;
			if ( CWTV.Site.featuredVideosSliderInstance ) {
				$("#homevideofeatured-wrap").css('width', videos_thumbs_width+'px');
			}

			if ( $('.videothumbslist').length ) {
				$(".videothumbslist").each(function(i) {
					var folder_index = $(this).data('index');
					var tc = CWTV.Site.videoPageSliderInstances[folder_index];
					if (tc) {
						var wrap_el = $("#touchcarousel_"+folder_index);
						wrap_el.css('width', videos_thumbs_width+'px');
					}
					
				});
			}

			for(var i=0; i<9; i++) {
				$('ul#day_'+i+' li div.wrap').attr('style', '');
				equalHeights('ul#day_'+i+' li div.wrap');
			}
		}
	}
}

	if (typeof initSeedVideo != 'undefined') {
		$('#video-container').show();
		var width, height;

		if ( container_width > 0 ) {
			width = container_width;
			height = Math.floor(width *  (405 / 720));
		} else {
			var win_h = $(window).height();
			height = win_h - 150;
			if ( height < 214 ) height = 214;
			width = 720 / 405 * height;
			if ( width < 178 ) width = 178;

		overlay_height = Math.floor(height - 210);
		if ( overlay_height < 0 ) overlay_height = 0;
		width+='px';
		height+='px';
		$('#videoplayer').css({'width':width,'height':height});	
		$('#videoplayer_flash').css({'width':width,'height':height});
		$('#vsplayer').css({'width':width,'height':height});
		$('#videoplayer_overlay').css({'height':overlay_height+'px'});
	}
}
