<script type="text/javascript">
  var onChangeInterval;
  var currentValue;
  var ajaxStatment;

  var autoComplete = {
      hightLight: function(direction) {

        //console.log("highlight called");

        var sel = $('a.ac-selected');

        //console.log(sel.length);

        if (sel.length == 0) {
            $('#nav_site .search_box .autocomplete .ac_section a:first').addClass("ac-selected");
        }
        else {

            var elementArray = new Array();

            $('#nav_site .search_box .autocomplete .ac_section a').each(function() {
                elementArray.push($(this));
            });

            var currentIndex = -1;

            for (i = 0; i < elementArray.length; i++) {
                var _this = elementArray[i];

                if ($(_this).hasClass("ac-selected")) {
                    //console.log("has class");
                    $(_this).removeClass("ac-selected");
                    //console.log("removing class")

                    if (direction == "up") {
                        //console.log("direction up is " + i);
                        currentIndex = i - 1;
                        //console.log("next index is " +currentIndex);
                    }
                    else {
                        //console.log("direction down");
                        currentIndex = i + 1;
                        //console.log("next index is " +currentIndex);
                    }

                    break;
                }
                
            }

            //console.log("current Index is => " + i);

            if (currentIndex != -1) {
                if (elementArray[currentIndex]) {
                    var selectedIndex = elementArray[currentIndex];
                    $(selectedIndex).addClass("ac-selected");
                }
            }
        }
      },
      checkKey: function(keynum) {
        var UP = 38;
        var DOWN = 40;
        var ENTER = 13;

        if (keynum === UP || keynum === DOWN) {
            return false;
        }
        else {
            return true;
        }
      },
      moveSelection: function(keynum, event) {
        var UP = 38;
        var DOWN = 40;
        var ENTER = 13;

        if(keynum === UP) {
           event.preventDefault();
           autoComplete.hightLight("up");
           _gaq.push(['_trackEvent', 'AutcompleteArrow', 'Click', "arrow-up"]);
        }

        if(keynum === DOWN) {
          event.preventDefault();  
          autoComplete.hightLight("down");
          _gaq.push(['_trackEvent', 'AutcompleteArrow', 'Click', "arrow-down"]);
        }

        if(keynum === ENTER) {
          if ($(".ac-selected").length != 0) {
            event.preventDefault();
            event.stopPropagation();
            _gaq.push(['_trackEvent', 'AutcompleteArrow', 'Click', "arrow-select"]);
            window.location.href = $(".ac-selected").attr('href');
          }
        }
      },
      checkValue: function(searchVal) {
          currentValue = searchVal;
      
          clearInterval(onChangeInterval);

          if (currentValue === '' || currentValue.length < 3) {
              //$(".brewery-complete").hide();
          } else {
              autoComplete.search(currentValue);
          }
      },
      validate: function(str) {
          return /^[a-z\d\-_\s]+$/i.test(str);
      },
      search: function(currentValue) {
          var d ="q="+currentValue;

          $(".autocomplete").attr('data-text', currentValue);

          /*if (ajaxStatment) {
              ajaxStatment.abort();
          }*/

          ajaxStatment = $.ajax({
              url: "/apireqs/autosearch",
              type: "GET",
              data: d,
              dataType: "json",
              error: function(xhr,ajaxOptions, thrownError)
              {
                  $(".autocomplete").removeClass("active");
              },
              success: function(data)
              {
                  if (data.error) {
                      $(".autocomplete").removeClass("active");
                  }
                  else {
                      if ($(".autocomplete").attr('data-text').length <= currentValue.length) {
                        var source   = $("#auto-complete-template").html();
                        var template = Handlebars.compile(source);
                        var searchResults = template(data.response);

                        $(".autocomplete").html(searchResults);
                        $(".autocomplete").addClass("active");
                      }
                  }
                  
              }
          });
      }
  }


 $(document).ready(function() {

    $("body").on("click", function(event) {
        if ($(".autocomplete").hasClass('active')) {
            if (event.currentTarget.className != "autocomplete") {
                $(".autocomplete").html('');
                $(".autocomplete").removeClass("active");
                $('.autocomplete .ac_section').removeClass(".ac-selected");
            }
        }

    });

    $(".search-input-desktop").on("keyup paste", function(event) {

        var searchVal = $(this).val();

        var keyCode = "";

        if (event.keyCode) { 
            keyCode = event.keyCode;
            autoComplete.moveSelection(event.keyCode, event);
        }  // IE
        else if(event.which) { 
            keyCode = event.which;
            autoComplete.moveSelection(event.which, event);
        }

        if (searchVal.length >= 2) {
            if (autoComplete.validate(searchVal)) {
              
              if (autoComplete.checkKey(keyCode)) {
                  if (searchVal == "") {
                      $(".autocomplete").html('');
                      $(".autocomplete").removeClass("active");
                  }

                  clearInterval(onChangeInterval);

                  if (currentValue !== searchVal);
                  {
                      onChangeInterval = setInterval(function() { autoComplete.checkValue(searchVal); }, 25);
                  }

                  //autoComplete.checkValue(searchVal);     
              }
          } 
        }

        
    });

    $(".track-focus").click(function() {
      var d = $(this).attr("data-track") + ":" + $(this).attr("placeholder");
      _gaq.push(['_trackEvent', 'Search Focus', 'Click', d]);
    });

    $(document).on("click", ".track-click", function() {
      var d = $(this).attr("data-track");
      if ($(this).attr("data-href")) {
        d = d + $(this).attr("data-href");
      }
      _gaq.push(['_trackEvent', 'Click', 'Click', d]);
    });
   });

  $(document).on("change", ".track-change", function() {
      var d = $(this).attr("data-track");
      if ($(this).attr("data-href")) {
        if ($(this).attr("data-href") == ":sort") {
          d = d + $(this).attr("data-href")+"/"+$(this).val();
        }
        else {
          d = d + $(this).attr("data-href");
        }
        
      }

      _gaq.push(['_trackEvent', 'Click', 'Click', d]);
  })

  function trackMobileEvent(label) {
    _gaq.push(['_trackEvent', 'Mobile Menu', 'Click', label]);
  }
</script>
