ytb.secondsToHOursAndMinutes = function(seconds){
	var minutes;
	var remainedSeconds;
	var time;
	var hours = Math.floor(seconds/3600);
	if(hours>0){
		minutes = Math.floor((seconds-hours*3600)/60);
		if(minutes>0){
			remainedSeconds=seconds - hours*3600 - minutes*60;
			time=hours+":"+minutes+":"+Math.floor(remainedSeconds);
			return time;
		}else {
			time=hours+":"+minutes;
			return time;
		}
	}else{
		minutes = Math.floor(seconds/60);
		remainedSeconds = seconds - minutes*60;
		time=minutes+":"+Math.floor(remainedSeconds);
		return time;
	}
}
