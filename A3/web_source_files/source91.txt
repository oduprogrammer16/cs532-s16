function updateCounter() {
    if (window.not_update_counter)
        return;
    
    $.get("/api/total_vipids_done/", function(nvideos) {
    
	var old = $("#videos-counter").attr("data-count");
	if (old != nvideos) {
            $("#videos-counter").attr("data-count", nvideos);
	    displayCount();
	}
    });
    
}

function displayCount() {
    var x = $("#videos-counter").attr("data-count");
    var formatted = x.toString().replace(/\B(?=(\d{3})+(?!\d))/g, ",");
    $("#videos-counter").text(formatted);
