(function ($) {
  // custom css expression for a case-insensitive contains()
  jQuery.expr[':'].Contains = function(a,i,m){
    return (a.textContent || a.innerText || "").replace(/[^a-zA-Z0-9]+/g, '').toUpperCase().indexOf(m[3].toUpperCase())>=0;
  };


  function listFilter(header, list) { // header is any element, list is an unordered list
    // create and add the filter form to the header
    var form = $(".filterform");
    var input = $(".filterinput");

    input
      .change( function () {
        var filter = $(this).val();
        filter = filter.replace(/[^a-zA-Z0-9]+/g, '');
        if (filter == "all") {
          $(list).find("li").show();
        } else if(filter) {
          // this finds all links in a list that contain the input,
          // and hide the ones not containing the input while showing the ones that do
          $(list).find("a:not(:Contains(" + filter + "))").parent().hide();
          $(list).find("a:Contains(" + filter + ")").parent().show();
        } else {
          $(list).find("li").hide();
        }
        if ($("#search-by-artist-content:visible").length) {
          var numrelated = $('#celebrity-list > li:visible').length;
          if (numrelated || !filter) {
            $("#no-results-found").addClass("hidden");
          } else {
            $("#no-results-found").removeClass("hidden");
          }
        }
        return false;
      })
    .keyup( function () {
        // fire the above change event after every letter
        $(this).change();
    });
  }


  //ondomready
  $(function () {
    listFilter($("#search-bar"), $("#celebrity-list"));
  });
}(jQuery));

$("#search-input").keydown(function(e) {
  switch(e.which) {
    case 38: // up
    if ($(".artist-hovered").length) {
      var cur = $(".artist-hovered");
      var prev = cur.parent().prevAll(":visible:first");
      if (prev.length) {
        $("a", prev).addClass("artist-hovered");
        if (prev.position().top < prev.height()) {
          var scrollTo = $("#celebrity-list").scrollTop() + prev.position().top;
          if (scrollTo < 0) {
            scrollTo = 0
          }
          $("#celebrity-list").scrollTop(scrollTo)
        }
        cur.removeClass("artist-hovered");
      }
    }
    break;

    case 40: // down
    if ($(".artist-hovered").length) {
      var cur = $(".artist-hovered");
      var next = cur.parent().nextAll(':visible:first');
      if (next.length) {
        $("a", next).addClass("artist-hovered");
        if (next.position().top > $("#celebrity-list").height() - next.height()) {
          $("#celebrity-list").scrollTop($("#celebrity-list").scrollTop() + next.height())
        }
        cur.removeClass("artist-hovered");
      }
    } else {
      $("a:visible:first", $("#search-by-artist-content")).addClass("artist-hovered");
    }
    break;

    case 13: // enter
    if ($(".artist-hovered").length) {
      var cur = $(".artist-hovered:first")[0];
      window.location = cur.href;
    }
    break;

    default: return; // exit this handler for other keys
  }
  e.preventDefault(); // prevent the default action (scroll / move caret)
});
