        function activate_gallery_mode_aberto(){
            $('.omelete-overlay').addClass('active');
            $('.right-panel.js-right-panel').show();
            $('.gallery').addClass('floating');
            var top = ($(window).height() - $('.gallery').height())/2;
            $('.gallery').css('top', top);

					  

					  $('.slick-current.slick-active').focus()[0].focus();
        }

        function deactivate_gallery_mode_aberto(){
            $('.omelete-overlay').removeClass('active');
            $('.right-panel.js-right-panel').hide();
            $('.gallery').removeClass('floating');
            $('.gallery').css('top', 0);
  					$('.js-menu').css('z-index', 110);
					  $('.slick-current.slick-active').focus()[0].focus();
					  $('.gallery-slider').attr('style', '');
					  $('.gallery-slider').resize();
        }

        function activate_fullscreen() {
            var $gallery = $('.gallery');
            $gallery.addClass('fullscreen');
            $gallery.css('top', 0);
            $('.gallery-item-big').height($('.gallery').height() - 70);
            $('.gallery-main').resize();
					  $('.gallery').css('top', top);

					  
					  $('.slick-current.slick-active').focus()[0].focus();
        }

        function deactivate_fullscreen() {
            var $gallery = $('.gallery');
            var $gallery_main = $('.gallery-main');

            $gallery.removeClass('fullscreen');
            $('.gallery-item-big').height(427);
            $gallery_main.resize();
            deactivate_gallery_mode_aberto();

            $('.gallery').removeClass('show-slider');
            $('.gallery-slider').removeClass('center-translateX');
            //$('.gallery-slider').width(732);
					  // $('.gallery-slider').width($('.gallery').width());

					  $('.gallery-slider').attr('style', '');
					  $('.gallery-slider').resize();

					  

            reload_main_slick();
            reload_slider_slick();
					  $('.slick-current.slick-active').focus()[0].focus();
        }

        function init_main_slick(slide){
            $('.gallery-main').slick({
                slidesToShow: 1,
                slidesToScroll: 1,
                arrows: false,
                fade: true,
                focusOnSelect: true,
                asNavFor: '.gallery-slider',
                lazyLoad: 'ondemand',
                initialSlide: slide,
							  adaptiveHeight: true,
            });
        }

        function init_slider_slick(slide) {
					var conf_dict = {
            slidesToShow: 7,
            slidesToScroll: 1,
            asNavFor: '.gallery-main',
            centerMode: true,
            focusOnSelect: true,
            lazyLoad: 'ondemand',
            centerPadding: '60px',
            initialSlide: slide,
            swipeToSlide: true,
          };

					var num_fotos = $('.gallery-slider .gallery-item').length;
					if( num_fotos <= 7) {
						conf_dict['centerMode'] = false;
						conf_dict['initialSlide'] = 0;
						conf_dict['slidesToShow'] = num_fotos;
					}
          $('.gallery-slider').slick(conf_dict);
					
					if( num_fotos <= 7) {
						/* 
						 * Select the correct slide on slick-slider when there is just a few photos
						 */
						$('.gallery-slider .gallery-item.slick-current').removeClass('slick-current');
						$('.gallery-slider .gallery-item[data-slick-index=' + slide + ']').addClass('slick-current');
					}
        }

        function reload_slider_slick(){
            var slide = $('.gallery-main').slick('slickCurrentSlide');
            $('.gallery-slider').slick('unslick');
            init_slider_slick(slide);
        }

        function reload_main_slick(){
            var slide = $('.gallery-main').slick('slickCurrentSlide');
            $('.gallery-main').slick('unslick');
            init_main_slick(slide);
        }

		 }
